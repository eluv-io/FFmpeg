202 /*!*****************************************************************************
203  * \brief Get time for logs with microsecond timestamps
204  *
205  * \param[in/out] p_tp timeval struct
206  * \param[in] p_tzp void *
207  *
208  * \return return 0 for success, -1 for error
209  ******************************************************************************/
210 int32_t ni_gettimeofday(struct timeval *p_tp, void *p_tzp)
211 {
212 #ifdef _WIN32
213  FILETIME file_time;
214  SYSTEMTIME system_time;
215  ULARGE_INTEGER ularge;
217  static const unsigned __int64 epoch =
218  ((unsigned __int64)116444736000000000ULL);
219 
220  // timezone information is stored outside the kernel so tzp isn't used
221  (void *)p_tzp;
223  // Note: this function is not a precision timer. See elapsed_time().
224  GetSystemTime(&system_time);
225  SystemTimeToFileTime(&system_time, &file_time);
226  ularge.LowPart = file_time.dwLowDateTime;
227  ularge.HighPart = file_time.dwHighDateTime;
228  // Surpress cppcheck
229  (void)ularge.LowPart;
230  (void)ularge.HighPart;
231  p_tp->tv_sec = (long)((ularge.QuadPart - epoch) / 10000000L);
232  p_tp->tv_usec = (long)(system_time.wMilliseconds * 1000);
233 
234  return 0;
235 #else
236  return gettimeofday(p_tp, p_tzp);
237 #endif
238 }

241 {

398 {
399 #ifdef _WIN32

401  {
402  LARGE_INTEGER StartCount;

404  LARGE_INTEGER Frequency;
405  QueryPerformanceCounter(&StartCount);
406  QueryPerformanceFrequency(&Frequency);
407  StopCount.QuadPart =
408  StartCount.QuadPart + usec * (Frequency.QuadPart / 1000000);
409  do
410  {
411  QueryPerformanceCounter(&StartCount);
412  } while (StartCount.QuadPart < StopCount.QuadPart);
413  } else
414  {
415  HANDLE timer = NULL;
416  LARGE_INTEGER ft = {0};
417  BOOL retval;
418 
419  ft.QuadPart = -(
420  10 *
421  usec); // Convert to 100 nanosecond interval, negative value indicates relative time
422 
423  timer = CreateWaitableTimer(NULL, TRUE, NULL);
424  if (NULL != timer)
425  {
426  retval = SetWaitableTimer(timer, &ft, 0, NULL, NULL, 0);
427  if (retval)
428  {
429  WaitForSingleObject(timer, INFINITE);
430  }
431  CloseHandle(timer);
432  } else
433  {
434  ni_log(NI_LOG_ERROR, "failed to create waitable timer\n");
435  LARGE_INTEGER StartCount;
436  LARGE_INTEGER StopCount;
437  LARGE_INTEGER Frequency;
438  QueryPerformanceCounter(&StartCount);
439  QueryPerformanceFrequency(&Frequency);
440  StopCount.QuadPart =

442  do

447  }

477 // get a free memory buffer from the pool
478 ni_buf_t *ni_buf_pool_get_buffer(ni_buf_pool_t *p_buffer_pool)
479 {
480  ni_buf_t *buf = NULL;
481 
482  if (NULL == p_buffer_pool)
483  {
484  return NULL;
485  }
486 
487  ni_pthread_mutex_lock(&p_buffer_pool->mutex);
488  buf = p_buffer_pool->p_free_head;
489 
490  // find and return a free buffer
491  if (NULL == buf)
492  {
493  ni_log(NI_LOG_INFO, "Expanding dec fme buffer_pool from %u to %u\n",
494  p_buffer_pool->number_of_buffers,
495  p_buffer_pool->number_of_buffers +
496  NI_DEC_FRAME_BUF_POOL_SIZE_EXPAND);
497 
498  buf = ni_buf_pool_expand(p_buffer_pool);
499  if (NULL == buf)
500  {
501  ni_pthread_mutex_unlock(&p_buffer_pool->mutex);
502  return NULL;
503  }
504  }
505 
506  // remove it from free list head; reconnect the linked list, the p_next
507  // will become the new head now
508  p_buffer_pool->p_free_head = buf->p_next_buffer;
509 
510  if (NULL != buf->p_next_buffer)
511  {
512  buf->p_next_buffer->p_previous_buffer = NULL;
513  } else
514  {
515  p_buffer_pool->p_free_tail = NULL;
516  }
517 
518  // add it to the used list tail
519  buf->p_previous_buffer = p_buffer_pool->p_used_tail;
520  buf->p_next_buffer = NULL;
521 
522  if (NULL != p_buffer_pool->p_used_tail)
523  {
524  p_buffer_pool->p_used_tail->p_next_buffer = buf;
525  } else
526  {
527  p_buffer_pool->p_used_head = buf;
528  }
529 
530  p_buffer_pool->p_used_tail = buf;
531 
532  ni_pthread_mutex_unlock(&p_buffer_pool->mutex);
533 
534  ni_log(NI_LOG_DEBUG, "%s ptr %p buf %p\n", __func__, buf->buf, buf);
535  return buf;
536 }

538 // return a used memory buffer to the pool
539 void ni_buf_pool_return_buffer(ni_buf_t *buf, ni_buf_pool_t *p_buffer_pool)
540 {
541  // p_buffer_pool could be null in case of delayed buffer return after pool
542  // has been freed
543  if (!buf)
544  {
545  return;
546  }
547 
548  ni_log(NI_LOG_DEBUG, "%s ptr %p buf %p\n", __func__, buf->buf, buf);
549 
550  if (!p_buffer_pool)
551  {
552  ni_log(NI_LOG_DEBUG, "%s: pool already freed, self destroy\n", __func__);
553  free(buf->buf);
554  free(buf);
555  return;
556  }
557 
558  ni_pthread_mutex_lock(&p_buffer_pool->mutex);
559 
560  // remove buf from the used list
561  if (NULL != buf->p_previous_buffer)
562  {
563  buf->p_previous_buffer->p_next_buffer = buf->p_next_buffer;
564  } else
565  {
566  p_buffer_pool->p_used_head = buf->p_next_buffer;
567  }
568 
569  if (NULL != buf->p_next_buffer)
570  {
571  buf->p_next_buffer->p_previous_buffer = buf->p_previous_buffer;
572  } else
573  {
574  p_buffer_pool->p_used_tail = buf->p_previous_buffer;
575  }
576 
577  // put it on the tail of free buffers list
578  buf->p_previous_buffer = p_buffer_pool->p_free_tail;
579  buf->p_next_buffer = NULL;
580 
581  if (NULL != p_buffer_pool->p_free_tail)
582  {
583  p_buffer_pool->p_free_tail->p_next_buffer = buf;
584  } else
585  {
586  p_buffer_pool->p_free_head = buf;
587  }
588 
589  p_buffer_pool->p_free_tail = buf;
590 
591  ni_pthread_mutex_unlock(&p_buffer_pool->mutex);
592 }

1076 /*!******************************************************************************
1077 * \brief Remove a string-pattern from a string in-place.
1078 *
1079 * \param[in,out] main_str Null terminated array of characters to operate upon in-place.
1080 * \param[in] pattern Null terminated array of characters to remove from main_str.
1081 * Supports special characters '#' and '+' for digit matching and
1082 * repeated matching respectively. Note, there is no way to \a escape
1083 * the special characters.
1084 * \b Example:
1085 * char a_str[10] = "aaa123qwe";

1090 * qwe

1092 * \return If characters removed, returns 1
1093 * If no characters removed, returns 0
1094 *******************************************************************************/
1095 uint32_t remove_substring_pattern(char *main_str, const char *pattern)
1096 {

1098  uint32_t match_length; // length of matching substring
1099  uint32_t matched_chr; // boolean flag for when match is found for a character in the pattern
1100  char char_match_pattern[11] = ""; // what characters to look for when evaluating a character in main_str
1101  uint32_t pattern_matched = 0; // boolean flag for when who pattern match is found
1102  uint32_t pattern_start = 0; // starting index in main_str of substring matching pattern
1103  const char digit_match_pattern[11] = "0123456789"; // set of numeric digits for expansion of special character '#'
1104 
1105  // do not accept zero length main_str or pattern
1106  if (!main_str || !pattern || !*main_str || !*pattern)
1107  {
1108  return 0;
1109  }
1110 
1111  // iterate over all characters in main_str looking for starting index of matching pattern
1112  for (i = 0; i < strlen(main_str) && !pattern_matched; i++)
1113  {
1114  pattern_matched = 0;
1115  match_length = 0;
1116  // iterate over the characters of the pattern
1117  for (j = 0; j < strlen(pattern); j++)
1118  {
1119  matched_chr = 0;
1120  // set which characters to look for, special cases for special control characters
1121  if (pattern[j] == '+')
1122  {
1123  // immediately fail as entering this branch means either the first character is a '+', or a '+" following a "+'
1124  return 0;
1125  }
1126  else if (pattern[j] == '#')
1127  {
1128  memcpy(char_match_pattern, digit_match_pattern, strlen(digit_match_pattern) + 1);
1129  }
1130  else
1131  {
1132  memcpy(char_match_pattern, pattern + j, 1);
1133  memset(char_match_pattern + 1, 0, 1);
1134  }
1135  // check if char is in match_pattern
1136  if (pattern[j+1] == '+')
1137  {
1138  while (main_str[i + match_length] && strchr(char_match_pattern, (int) main_str[i + match_length]))
1139  {
1140  match_length++;
1141  matched_chr = 1;
1142  }
1143  j++;
1144  }
1145  else if (main_str[i + match_length] && strchr(char_match_pattern, (int) main_str[i + match_length]))
1146  {
1147  match_length++;
1148  matched_chr = 1;
1149  }
1150  // if no matches were found, then this segment is not the sought pattern
1151  if (!matched_chr)
1152  {
1153  break;
1154  }
1155  // if all of pattern has been processed and matched, mark sucessful whole pattern match
1156  else if ((j + 1) >= strlen(pattern))
1157  {
1158  pattern_matched = 1;
1159  pattern_start = i;
1160  }
1161  }
1162  }
1163 
1164  // remove sub-string if its pattern was found in main_str
1165  if (pattern_matched)
1166  {
1167  uint32_t orig_main_str_len = (uint32_t)strlen(main_str);
1168  memmove(main_str + pattern_start, main_str + pattern_start + match_length,
1169  strlen(main_str + pattern_start + match_length));
1170  main_str[orig_main_str_len - match_length] = 0;

1172  }
1173  else
1174  {
1175  return 0;
1176  }
