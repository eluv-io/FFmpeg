71 /*!*****************************************************************************
72  * \brief write a byte to bitstream
73  * Note: the stream must be byte-aligned already
74  *
75  * \param stream bitstream
76  * \param byte byte to write
77  * \return none
78  ******************************************************************************/
79 static void ni_bs_writer_write_byte(ni_bitstream_writer_t *stream, uint8_t byte)
80 {
81  assert(stream->cur_bit == 0);
82 
83  if (stream->last == NULL || stream->last->len == NI_DATA_CHUNK_SIZE)
84  {
85  // need to allocate a new chunk.
86  ni_data_chunk_t *new_chunk = ni_bs_writer_alloc_chunk();
87  if (!new_chunk)
88  {
89  ni_log(NI_LOG_ERROR, "%s error: no memory\n", __func__);
90  return;
91  }
92 
93  if (!stream->first)
94  stream->first = new_chunk;
95  if (stream->last)
96  stream->last->next = new_chunk;
97  stream->last = new_chunk;
98  }
99  if (stream->last->len >= NI_DATA_CHUNK_SIZE)
100  {
101  ni_log(NI_LOG_ERROR, "%s error: new_chunk size >= max %d\n", __func__,
102  NI_DATA_CHUNK_SIZE);
103  return;
104  }
105 
106  stream->last->data[stream->last->len] = byte;
107  stream->last->len += 1;
108  stream->len += 1;
109 }

127 static inline unsigned ni_math_floor_log2(unsigned value)
128 {
129  unsigned result = 0;
130  assert(value > 0);
131  int i;
132 
133  for (i = 4; i >= 0; --i)
134  {
135  unsigned bits = 1ull << i;
136  unsigned shift = (value >= (1ull << bits)) ? bits : 0;
137  result += shift;
138  value >>= shift;
139  }
140 
141  return result;
142 }

144 static inline unsigned ni_math_ceil_log2(unsigned value)
145 {
146  assert(value > 0);
147 
148  // the ceil_log2 is just floor_log2 + 1, except for exact powers of 2.
149  return ni_math_floor_log2(value) + ((value & (value - 1)) ? 1 : 0);
150 }

175 /*!*****************************************************************************
176  * \brief write a specified number (<= 32) of bits to bitstream,
177  * buffer individual bits until a full byte is made
178  * \param stream bitstream
179  * \param data input data
180  * \param bits number of bits in data to write to stream, max 32
181  * \return none
182  ******************************************************************************/
183 void ni_bs_writer_put(ni_bitstream_writer_t *stream, uint32_t data,
184  uint8_t bits)
185 {
186  if (bits > 32)
187  {
188  ni_log(NI_LOG_ERROR, "%s error: too many bits to write: %u\n", __func__,
189  bits);
190  return;
191  }
192 
193  while (bits--)
194  {
195  stream->data <<= 1;
196 
197  if (data & ni_bit_set_mask[bits])
198  {
199  stream->data |= 1;
200  }
201  stream->cur_bit++;
202 
203  // write the complete byte
204  if (stream->cur_bit == 8)
205  {
206  stream->cur_bit = 0;
207  ni_bs_writer_write_byte(stream, stream->data);
208  }
209  }
210 }

212 /*!*****************************************************************************
213  * \brief write unsigned Exp-Golomb bit string to bitstream
214  *
215  * \param stream bitstream
216  * \param data input data
217  * \return none
218  ******************************************************************************/
219 void ni_bs_writer_put_ue(ni_bitstream_writer_t *stream, uint32_t data)
220 {
221  unsigned data_log2 = ni_math_floor_log2(data + 1);
222  unsigned prefix = 1 << data_log2;
223  unsigned suffix = data + 1 - prefix;
224  unsigned num_bits = data_log2 * 2 + 1;
225  unsigned value = prefix | suffix;
226 
227  ni_bs_writer_put(stream, value, num_bits);
228 }

234  * \param data input data
235  * \return none
236  ******************************************************************************/
237 void ni_bitstream_put_se(ni_bitstream_writer_t *stream, int32_t data)
238 {
239  // map positive value to even and negative to odd value
240  uint32_t data_num = data <= 0 ? (-data) << 1 : (data << 1) - 1;
241  ni_bs_writer_put_ue(stream, data_num);

345 void ni_bs_reader_skip_bits(ni_bitstream_reader_t *br, int n)
346 {
347  int new_offset = 8 * br->byte_offset + br->bit_offset + n;
348  if (new_offset > br->size_in_bits)
349  {

351  "%s: skip %d, current byte_offset "
352  "%d bit_offset %d, over total size %d, stop !\n",
353  __func__, n, br->byte_offset, br->bit_offset, br->size_in_bits);
354  return;
355  }
356 
357  br->byte_offset = new_offset / 8;
358  br->bit_offset = new_offset % 8;
359 }

361 // read a single bit
362 uint8_t ni_bitstream_get_1bit(ni_bitstream_reader_t *br)
363 {
364  uint8_t ret = 0;
365 
366  if (0 == br->bit_offset)
367  {
368  ret = (br->buf[br->byte_offset] >> 7);
369  br->bit_offset = 1;
370  } else
371  {
372  ret = ((br->buf[br->byte_offset] >> (7 - br->bit_offset)) & 0x1);
373  if (7 == br->bit_offset)
374  {
375  br->bit_offset = 0;
376  br->byte_offset++;
377  } else
378  {
379  br->bit_offset++;
380  }
381  }
382 
383  return ret;
384 }

402 // read a 16 bit integer
403 uint16_t ni_bitstream_get_u16(ni_bitstream_reader_t *br)
404 {
405  uint16_t ret = 0;
406  int i;
407  int offset;
408  const uint8_t *src = NULL;
409 
410  src = &(br->buf[br->byte_offset]);
411  offset = 16 + br->bit_offset;
412 
413  for (i = 0; i < 2; i++)
414  {
415  offset -= 8;
416  ret |= ((uint16_t)src[i] << offset);
417  }
418 
419  if (0 != offset)
420  {
421  ret |= (src[2] >> (8 - offset));
422  }
423 
424  br->byte_offset += 2;
425 
426  return ret;
427 }

448 /*!*****************************************************************************
449  * \brief read bits (up to 32) from the bitstream reader, after reader init
450  *
451  * \param br bitstream reader
452  * \param n number of bits to read
453  * \return value read
454  ******************************************************************************/
455 uint32_t ni_bs_reader_get_bits(ni_bitstream_reader_t *br, int n)
456 {
457  uint32_t ret = 0;
458  int bits_left;
459 
460  if (n > 32)
461  {
462  ni_log(NI_LOG_ERROR, "%s %d bits > 32, not supported!\n", __func__, n);
463  return 0;
464  }
465 
466  if (n <= 0)
467  {
468  // return 0
469  } else if (n < 8)
470  {
471  ret = ni_bitstream_get_8bits_or_less(br, n);
472  } else if (8 == n)
473  {
474  ret = ni_bitstream_get_u8(br);
475  } else if (n > 8 && n < 16)
476  {
477  bits_left = n % 8;
478  ret = ((ni_bitstream_get_8bits_or_less(br, bits_left) << 8) |
479  ni_bitstream_get_u8(br));
480  } else if (16 == n)
481  {
482  ret = ni_bitstream_get_u16(br);
483  } else if (n > 16 && n < 24)
484  {
485  bits_left = n % 16;
486  ret = (ni_bitstream_get_8bits_or_less(br, bits_left) << 16);
487  ret |= (ni_bitstream_get_u8(br) << 8);
488  ret |= ni_bitstream_get_u8(br);
489  } else // 32 >= n >= 24
490  {
491  bits_left = n % 24;
492  ret = (ni_bitstream_get_8bits_or_less(br, bits_left) << 24);
493  ret |= (ni_bitstream_get_u8(br) << 16);
494  ret |= (ni_bitstream_get_u8(br) << 8);
495  ret |= ni_bitstream_get_u8(br);
496  }
497  return ret;
498 }

500 /*!*****************************************************************************
501  * \brief read an unsigned Exp-Golomb code ue(v)
502  *
503  * \param br bitstream reader
504  * \return value read
505  ******************************************************************************/
506 uint32_t ni_bs_reader_get_ue(ni_bitstream_reader_t *br)
507 {
508  uint32_t ret = 0;
509  int i = 0; // leading zero bits
510 
511  // count leading zero bits
512  while (0 == ni_bitstream_get_1bit(br) && i < 32)
513  {
514  i++;
515  }
516  if (i == 32)
517  return 0;
518  // calc get_bits(leading zero bits)
519  ret = ni_bs_reader_get_bits(br, i);
520  ret += (1U << i) - 1;
521  return ret;
522 }
