52 /*!******************************************************************************
53  * \brief get the NVMe device's character device name (e.g. /dev/nvmeX)
54  *
55  * \param in whole device name passed in
56  * dev_name full device name without name-space/partition-number
57  *
58  * \return 0 if value device name is found, -1 otherwise
59  ******************************************************************************/
60 static int get_dev_name(const char *in, char *dev_name)
61 {
62  char *tmp = NULL;
63 
64  if ( dev_name && (strlen(in) > strlen(DEV_NAME_PREFIX) && strstr(in, DEV_NAME_PREFIX)) )
65  {
66  tmp = (char *)in + strlen(DEV_NAME_PREFIX);
67  while (isdigit(*tmp))
68  tmp++;
69  strncpy(dev_name, in, tmp - in);
70  dev_name[tmp - in] = '\0';
71  return 0;
72  }
73  return -1;
74 }

98 int main(int argc, char *argv[])

108  if (argc == 1) {

113  // arg handling

115  {

151  optarg);

154  break;
155  case 'v':
156  printf("Release ver: %s\n"
157  "API ver: %s\n"
158  "Date: %s\n"
159  "ID: %s\n",
160  NI_XCODER_REVISION, LIBXCODER_API_VERSION,

162  return 0;
163  case 'h':
164  default:
165  display_help();
166  return 0;
167  }
168  }
169 
170  // check option
171  if (add_dev && (del_dev || delete_all))
172  {

174  display_help();
175  return 1;
176  }
177  if (!should_match_rev && !add_dev)
178  {
179  fprintf(stderr, "Error: -r option must be used with -a option\n\n");
180  display_help();
181  return 1;
182  }
183  if (add_dev)
184  {
185  rc = ni_rsrc_add_device(char_dev_name, should_match_rev);
186  if (rc)
187  printf("%s not added as transcoder.\n", char_dev_name);
188  else
189  printf("Added transcoder %s successfully.\n", char_dev_name);
190  return rc;
191  }
192  else if (delete_all)
193  {
194  rc = ni_rsrc_remove_all_devices();
195  if (rc)
196  printf("Error removing all transcoder resources.\n");
197  else
198  printf("Removing all transcoder resources successfully.\n");
199  return rc;
200  }
201  else if (del_dev)
202  {
203  rc = ni_rsrc_remove_device(char_dev_name);
204  if (rc)
205  printf("%s not removed as transcoder.\n", char_dev_name);
206  else
207  printf("Removed transcoder %s successfully.\n", char_dev_name);
208  return rc;
209  }
210  else
211  {
212  fprintf(stderr, "Error: ni_rsrc_update option must be used with -a or -b or -D option\n\n");
213  display_help();
214  return 1;
215  }
216 }
