89 /*!****************************************************************************
90  * \brief Read the next frame
91  *
92  * \param[in] fd file descriptor of input file
93  * \param[out] p_dst pointer to place the frame
94  * \param[in] to_read number of bytes to copy to the pointer
95  *
96  * \return bytes copied
97  ******************************************************************************/
98 int read_next_chunk_from_file(int fd, uint8_t *p_dst, uint32_t to_read)
99 {
100  uint8_t *tmp_dst = p_dst;
101  ni_log(NI_LOG_DEBUG,
102  "read_next_chunk_from_file:p_dst %p len %u totalSize %llu left %llu\n",
103  tmp_dst, to_read, (unsigned long long)total_file_size, (unsigned long long)data_left_size);
104  int to_copy = to_read;
105  unsigned long tmpFileSize = to_read;
106  if (data_left_size == 0)
107  {
108  if (g_repeat > 1)
109  {
110  data_left_size = total_file_size;
111  g_repeat--;
112  ni_log(NI_LOG_DEBUG, "input processed %d left\n", g_repeat);
113  lseek(fd, 0, SEEK_SET); //back to beginning
114  } else
115  {
116  return 0;
117  }
118  } else if (data_left_size < to_read)
119  {
120  tmpFileSize = data_left_size;
121  to_copy = data_left_size;
122  }
123 
124  int one_read_size = read(fd, tmp_dst, to_copy);
125  if (one_read_size == -1)
126  {
127  fprintf(stderr, "Error: reading file, quit! left-to-read %lu\n",
128  tmpFileSize);
129  fprintf(stderr, "Error: input file read error\n");
130  return -1;
131  }
132  data_left_size -= one_read_size;
133 
134  return to_copy;
135 }

197 /*!****************************************************************************
198  * \brief Reads YUV data from input file then calls a special libxcoder API
199  * function to transfer the YUV data into the hardware frame on
200  * the Quadra device.
201  *
202  * \param [in] p_upl_ctx pointer to upload session context
203  * [in] fd file descriptor of input file
204  * [in] p_yuv420p_frame address of pointer to YUV data
205  * [in] p_in_frame pointer to hardware frame
206  * [in] input_video_width video width
207  * [in] input_video_height video height
208  * [out] bytes_sent updated byte count of total data read
209  * [out] input_exhausted set to 1 when we reach end-of-file
210  *
211  * \return 0 on success
212  * -1 on error
213  ******************************************************************************/
214 int p2p_upload_send_data(ni_session_context_t *p_upl_ctx, int fd,
215  uint8_t **p_yuv420p_frame, ni_frame_t *p_in_frame,
216  int input_video_width, int input_video_height,
217  unsigned long *bytes_sent, int *input_exhausted)
218 {
219  static uint8_t tmp_buf[MAX_YUV_FRAME_SIZE];
220  void *p_buffer;
221  uint8_t *p_src[NI_MAX_NUM_DATA_POINTERS];
222  uint8_t *p_dst[NI_MAX_NUM_DATA_POINTERS];
223  int src_stride[NI_MAX_NUM_DATA_POINTERS];
224  int src_height[NI_MAX_NUM_DATA_POINTERS];
225  int dst_stride[NI_MAX_NUM_DATA_POINTERS] = {0, 0, 0, 0};
226  int dst_height[NI_MAX_NUM_DATA_POINTERS] = {0, 0, 0, 0};
227  int frame_size;
228  int chunk_size;
229  int alignedh;
230  int Ysize;
231  int Usize;
232  int Vsize;
233  int total_size;
234 
235  ni_log(NI_LOG_DEBUG, "===> p2p upload_send_data <===\n");
236 
237  /* An 8-bit YUV420 planar frame occupies [(width x height x 3)/2] bytes */
238  frame_size = input_video_height * input_video_width * 3 / 2;
239 
240  chunk_size = read_next_chunk_from_file(fd, tmp_buf, frame_size);
241 
242  if (chunk_size == 0)
243  {
244  ni_log(NI_LOG_DEBUG, "p2p_upload_send_data: read chunk size 0, eos!\n");
245  *input_exhausted = 1;
246  }
247 
248  p_in_frame->video_width = input_video_width;
249  p_in_frame->video_height = input_video_height;
250  p_in_frame->extra_data_len = 0;
251 
252  ni_get_hw_yuv420p_dim(input_video_width, input_video_height,
253  p_upl_ctx->bit_depth_factor, 0, dst_stride,
254  dst_height);
255 
256  ni_log(NI_LOG_DEBUG, "p_dst alloc linesize = %d/%d/%d src height=%d "
257  "dst height aligned = %d/%d/%d \n",
258  dst_stride[0], dst_stride[1], dst_stride[2],
259  input_video_height, dst_height[0], dst_height[1],
260  dst_height[2]);
261 
262  src_stride[0] = input_video_width * p_upl_ctx->bit_depth_factor;
263  src_stride[1] = src_stride[2] = src_stride[0] / 2;
264 
265  src_height[0] = input_video_height;
266  src_height[1] = src_height[0] / 2;
267  src_height[2] = src_height[1];
268 
269  p_src[0] = tmp_buf;
270  p_src[1] = tmp_buf + src_stride[0] * src_height[0];
271  p_src[2] = p_src[1] + src_stride[1] * src_height[1];
272 
273  alignedh = (input_video_height + 1) & ~1;
274 
275  Ysize = dst_stride[0] * alignedh;
276  Usize = dst_stride[1] * alignedh / 2;
277  Vsize = dst_stride[2] * alignedh / 2;
278 
279  total_size = Ysize + Usize + Vsize;
280  total_size = ((total_size + 4095) & ~4095) + 4096;
281 
282  if (*p_yuv420p_frame == NULL)
283  {
284  if (ni_posix_memalign(&p_buffer, sysconf(_SC_PAGESIZE), total_size))
285  {
286  fprintf(stderr, "Can't alloc memory\n");
287  return -1;
288  }
289 
290  *p_yuv420p_frame = p_buffer;
291  }
292 
293  p_dst[0] = *p_yuv420p_frame;
294  p_dst[1] = *p_yuv420p_frame + Ysize;
295  p_dst[2] = *p_yuv420p_frame + Ysize + Usize;
296  p_dst[3] = NULL;
297 
298  ni_copy_hw_yuv420p(p_dst, p_src, input_video_width, input_video_height, 1,
299  0, 0, dst_stride, dst_height, src_stride, src_height);
300 #ifndef _WIN32
301  if (ni_uploader_p2p_test_send(p_upl_ctx, *p_yuv420p_frame, total_size,
302  p_in_frame))
303  {
304  fprintf(stderr, "Error: failed ni_uploader_p2p_test_send()\n");
305  return -1;
306  } else
307 #endif
308  {
309  *bytes_sent = total_size;
310  }
311 
312  return 0;
313 }

315 /*!****************************************************************************
316  * \brief Prepare frames to simulate P2P transfers
317  *
318  * \param [in] p_upl_ctx pointer to caller allocated uploader
319  * session context
320  * [in] input_video_width video width
321  * [in] input_video_height video height
322  * [out] p2p_frame array of hw frames
323  *
324  * \return 0 on success
325  * -1 on error
326  ******************************************************************************/
327 int p2p_prepare_frames(ni_session_context_t *p_upl_ctx, int input_video_width,
328  int input_video_height, ni_frame_t p2p_frame[])
329 {
330  int i;
331  int ret = 0;
332  int dst_stride[NI_MAX_NUM_DATA_POINTERS] = {0, 0, 0, 0};
333  int dst_height[NI_MAX_NUM_DATA_POINTERS] = {0, 0, 0, 0};
334  ni_frame_t *p_in_frame;
335 
336  // Allocate memory for two hardware frames
337  for (i = 0; i < POOL_SIZE; i++)
338  {
339  p_in_frame = &p2p_frame[i];
340 
341  p_in_frame->start_of_stream = 0;
342  p_in_frame->end_of_stream = 0;
343  p_in_frame->force_key_frame = 0;
344  p_in_frame->extra_data_len = 0;
345 
346  ni_get_hw_yuv420p_dim(input_video_width, input_video_height,
347  p_upl_ctx->bit_depth_factor, 0, dst_stride,
348  dst_height);
349 
350  // Allocate a hardware ni_frame structure for the encoder
351  if (ni_frame_buffer_alloc_hwenc(
352  p_in_frame, input_video_width, input_video_height,
353  (int)p_in_frame->extra_data_len) != NI_RETCODE_SUCCESS)
354  {
355  fprintf(stderr, "Error: could not allocate hw frame buffer!");
356  ret = -1;
357  goto fail_out;
358  }
359 
360 #ifndef _WIN32
361  // Acquire a hw frame from the upload session. This obtains a handle
362  // to Quadra memory from the previously created frame pool.
363  if (ni_device_session_acquire(p_upl_ctx, p_in_frame))
364  {
365  fprintf(stderr, "Error: failed ni_device_session_acquire()\n");
366  ret = -1;
367  goto fail_out;
368  }
369 #endif
370  }
371 
372  return ret;
373 
374 fail_out:
375  for (i = 0; i < POOL_SIZE; i++)
376  {
377  ni_frame_buffer_free(&(p2p_frame[i]));
378  }
379 
380  return ret;
381 }

383 /*!****************************************************************************
384  * \brief Send the Quadra encoder a hardware frame which triggers
385  * Quadra to encode the frame
386  *
387  * \param [in] p_enc_ctx pointer to encoder context
388  * [in] p_in_frame pointer to hw frame
389  * [in] input_exhausted flag indicating this is the last frame
390  * [in/out] need_to_resend flag indicating need to re-send
391  *
392  * \return 0 on success
393  * -1 on failure
394  ******************************************************************************/
395 int encoder_encode_frame(ni_session_context_t *p_enc_ctx,
396  ni_frame_t *p_in_frame, int input_exhausted,
397  int *need_to_resend)
398 {
399  static int started = 0;
400  int oneSent;
401  ni_session_data_io_t in_data;
402 
403  ni_log(NI_LOG_DEBUG, "===> encoder_encode_frame <===\n");
404 
405  if (enc_eos_sent == 1)
406  {
407  ni_log(NI_LOG_DEBUG, "encoder_encode_frame: ALL data (incl. eos) sent "
408  "already!\n");
409  return 0;
410  }
411 
412  if (*need_to_resend)
413  {
414  goto send_frame;
415  }
416 
417  p_in_frame->start_of_stream = 0;
418 
419  // If this is the first frame, mark the frame as start-of-stream
420  if (!started)
421  {
422  started = 1;
423  p_in_frame->start_of_stream = 1;
424  }
425 
426  // If this is the last frame, mark the frame as end-of-stream
427  p_in_frame->end_of_stream = input_exhausted ? 1 : 0;
428  p_in_frame->force_key_frame = 0;
429 
430 send_frame:
431 
432  in_data.data.frame = *p_in_frame;
433  oneSent =
434  ni_device_session_write(p_enc_ctx, &in_data, NI_DEVICE_TYPE_ENCODER);
435 
436  if (oneSent < 0)
437  {
438  fprintf(stderr,
439  "Error: failed ni_device_session_write() for encoder\n");
440  *need_to_resend = 1;
441  return -1;
442  } else if (oneSent == 0 && !p_enc_ctx->ready_to_close)
443  {
444  *need_to_resend = 1;
445  ni_log(NI_LOG_DEBUG, "NEEDED TO RESEND");
446  } else
447  {
448  *need_to_resend = 0;
449 
450  ni_log(NI_LOG_DEBUG, "encoder_encode_frame: total sent data size=%u\n",
451  p_in_frame->data_len[3]);
452 
453  ni_log(NI_LOG_DEBUG, "encoder_encode_frame: success\n");
454 
455  if (p_enc_ctx->ready_to_close)
456  {
457  enc_eos_sent = 1;
458  }
459  }
460 
461  return 0;
462 }

464 /*!****************************************************************************
465  * \brief Receive output packet data from the Quadra encoder
466  *
467  * \param [in] p_enc_ctx pointer to encoder session context
468  * [in] p_out_data pointer to output data session
469  * [in] p_file pointer to file to write the packet
470  * [out] total_bytes_received running counter of bytes read
471  * [in] print_time 1 = print the time
472  *
473  * \return 0 - success got packet
474  * 1 - received eos
475  * 2 - got nothing, need retry
476  * -1 - failure
477  ******************************************************************************/
478 int encoder_receive_data(ni_session_context_t *p_enc_ctx,
479  ni_session_data_io_t *p_out_data, FILE *p_file,
480  unsigned long long *total_bytes_received,
481  int print_time)
482 {
483  int packet_size = NI_MAX_TX_SZ;
484  int rc = 0;
485  int end_flag = 0;
486  int rx_size = 0;
487  int meta_size = p_enc_ctx->meta_size;
488  ni_packet_t *p_out_pkt = &(p_out_data->data.packet);
489  static int received_stream_header = 0;
490 
491  ni_log(NI_LOG_DEBUG, "===> encoder_receive_data <===\n");
492 
493  if (NI_INVALID_SESSION_ID == p_enc_ctx->session_id ||
494  NI_INVALID_DEVICE_HANDLE == p_enc_ctx->blk_io_handle)
495  {
496  ni_log(NI_LOG_DEBUG, "encode session not opened yet, return\n");
497  return 0;
498  }
499 
500  if (p_file == NULL)
501  {
502  ni_log(NI_LOG_ERROR, "Bad file pointer, return\n");
503  return -1;
504  }
505 
506  rc = ni_packet_buffer_alloc(p_out_pkt, packet_size);
507  if (rc != NI_RETCODE_SUCCESS)
508  {
509  fprintf(stderr, "Error: malloc packet failed, ret = %d!\n", rc);
510  return -1;
511  }
512 
513  /*
514  * The first data read from the encoder session context
515  * is a stream header read.
516  */
517  if (!received_stream_header)
518  {
519  /* Read the encoded stream header */
520  rc = ni_encoder_session_read_stream_header(p_enc_ctx, p_out_data);
521 
522  if (rc > 0)
523  {
524  /* Write out the stream header */
525  if (fwrite((uint8_t *)p_out_pkt->p_data + meta_size,
526  p_out_pkt->data_len - meta_size, 1, p_file) != 1)
527  {
528  fprintf(stderr, "Error: writing data %u bytes error!\n",
529  p_out_pkt->data_len - meta_size);
530  fprintf(stderr, "Error: ferror rc = %d\n", ferror(p_file));
531  }
532 
533  *total_bytes_received += (rx_size - meta_size);
534  number_of_packets++;
535  received_stream_header = 1;
536  } else if (rc != 0)
537  {
538  fprintf(stderr, "Error: reading header %d\n", rc);
539  return -1;
540  }
541 
542  if (print_time)
543  {
544  int timeDiff = (int)(current_time.tv_sec - start_time.tv_sec);
545  if (timeDiff == 0)
546  {
547  timeDiff = 1;
548  }
549  printf("[R] Got:%d Packets= %u fps=%u Total bytes %llu\n",
550  rx_size, number_of_packets, number_of_packets / timeDiff,
551  *total_bytes_received);
552  }
553 
554  /* This shouldn't happen */
555  if (p_out_pkt->end_of_stream)
556  {
557  return 1;
558  } else if (rc == 0)
559  {
560  return 2;
561  }
562  }
563 
564 receive_data:
565  rc = ni_device_session_read(p_enc_ctx, p_out_data, NI_DEVICE_TYPE_ENCODER);
566 
567  end_flag = p_out_pkt->end_of_stream;
568  rx_size = rc;
569 
570  ni_log(NI_LOG_DEBUG, "encoder_receive_data: received data size=%d\n", rx_size);
571 
572  if (rx_size > meta_size)
573  {
574  if (fwrite((uint8_t *)p_out_pkt->p_data + meta_size,
575  p_out_pkt->data_len - meta_size, 1, p_file) != 1)
576  {
577  fprintf(stderr, "Error: writing data %u bytes error!\n",
578  p_out_pkt->data_len - meta_size);
579  fprintf(stderr, "Error: ferror rc = %d\n", ferror(p_file));
580  }
581 
582  *total_bytes_received += rx_size - meta_size;
583  number_of_packets++;
584 
585  ni_log(NI_LOG_DEBUG, "Got: Packets= %u\n", number_of_packets);
586  } else if (rx_size != 0)
587  {
588  fprintf(stderr, "Error: received %d bytes, <= metadata size %d!\n",
589  rx_size, meta_size);
590  return -1;
591  } else if (!end_flag &&
592  ((ni_xcoder_params_t *)(p_enc_ctx->p_session_config))
593  ->low_delay_mode)
594  {
595  ni_log(NI_LOG_DEBUG, "low delay mode and NO pkt, keep reading...\n");
596  goto receive_data;
597  }
598 
599  if (print_time)
600  {
601  int timeDiff = (int)(current_time.tv_sec - start_time.tv_sec);
602  if (timeDiff == 0)
603  {
604  timeDiff = 1;
605  }
606  printf("[R] Got:%d Packets= %u fps=%u Total bytes %llu\n", rx_size,
607  number_of_packets, number_of_packets / timeDiff,
608  *total_bytes_received);
609  }
610 
611  if (end_flag)
612  {
613  printf("Encoder Receiving done\n");
614  return 1;
615  } else if (0 == rx_size)
616  {
617  return 2;
618  }
619 
620  ni_log(NI_LOG_DEBUG, "encoder_receive_data: success\n");
621 
622  return 0;
623 }

625 /*!****************************************************************************
626  * \brief Open an encoder session to Quadra
627  *
628  * \param [out] p_enc_ctx pointer to an encoder session context
629  * [in] dst_codec_format AVC or HEVC
630  * [in] iXcoderGUID id to identify the Quadra device
631  * [in] p_enc_params sets the encoder parameters
632  * [in] width width of frames to encode
633  * [in] height height of frames to encode
634  *
635  * \return 0 if successful, < 0 otherwise
636  ******************************************************************************/
637 int encoder_open_session(ni_session_context_t *p_enc_ctx, int dst_codec_format,
638  int iXcoderGUID, ni_xcoder_params_t *p_enc_params,
639  int width, int height, ni_frame_t *p_frame)
640 {
641  int ret = 0;
642 
643  // Enable hardware frame encoding
644  p_enc_ctx->hw_action = NI_CODEC_HW_ENABLE;
645  p_enc_params->hwframes = 1;
646 
647  // Provide the first frame to the Quadra encoder
648  p_enc_params->p_first_frame = p_frame;
649 
650  // Specify codec, AVC vs HEVC
651  p_enc_ctx->codec_format = dst_codec_format;
652 
653  p_enc_ctx->p_session_config = p_enc_params;

770 /*!****************************************************************************
771  * \brief Parse user command line arguments
772  *
773  * \param [in] argc argument count
774  * [in] argv argument vector
775  * [out] input_filename input filename
776  * [out] output_filename output filename
777  * [out] iXcoderGUID Quadra device

784 void parse_arguments(int argc, char *argv[], char *input_filename,
785  char *output_filename, int *iXcoderGUID, int *arg_width,
786  int *arg_height, int *dst_codec_format)
787 {
788  char xcoderGUID[32];
789  char mode_description[128];
790  char *n; // used for parsing width and height from --size
791  size_t i;
792  int opt;
793  int opt_index;
794  ni_log_level_t log_level;
795 
796  static const char *opt_string = "hvl:c:i:s:m:o:r:";
797  static const struct option long_options[] = {
798  {"help", no_argument, NULL, 'h'},
799  {"version", no_argument, NULL, 'v'},
800  {"loglevel", no_argument, NULL, 'l'},
801  {"card", required_argument, NULL, 'c'},
802  {"input", required_argument, NULL, 'i'},
803  {"size", required_argument, NULL, 's'},
804  {"mode", required_argument, NULL, 'm'},
805  {"output", required_argument, NULL, 'o'},
806  {"repeat", required_argument, NULL, 'r'},
807  {NULL, 0, NULL, 0},
808  };
809 
810  while ((opt = getopt_long(argc, argv, opt_string, long_options,
811  &opt_index)) != -1)
812  {
813  switch (opt)
814  {
815  case 'h':
816  print_usage();
817  exit(0);
818  case 'v':
819  printf("Release ver: %s\n"
820  "API ver: %s\n"
821  "Date: %s\n"
822  "ID: %s\n",
823  NI_XCODER_REVISION, LIBXCODER_API_VERSION,
824  NI_SW_RELEASE_TIME, NI_SW_RELEASE_ID);
825  exit(0);
826  case 'l':
827  log_level = arg_to_ni_log_level(optarg);
828  if (log_level != NI_LOG_INVALID)
829  {
830  ni_log_set_level(log_level);
831  } else {
832  arg_error_exit("-l | --loglevel", optarg);
833  }
834  break;
835  case 'c':
836  strcpy(xcoderGUID, optarg);
837  *iXcoderGUID = (int)strtol(optarg, &n, 10);
838  // No numeric characters found in left side of optarg
839  if (n == xcoderGUID)
840  arg_error_exit("-c | --card", optarg);
841  break;
842  case 'i':
843  strcpy(input_filename, optarg);
844  break;
845  case 's':
846  *arg_width = (int)strtol(optarg, &n, 10);
847  *arg_height = atoi(n + 1);
848  if ((*n != 'x') || (!*arg_width || !*arg_height))
849  arg_error_exit("-s | --size", optarg);
850  break;
851  case 'm':
852  if (!(strlen(optarg) == 3))
853  arg_error_exit("-m | --mode", optarg);
854 
855  // convert to lower case for processing
856  for (i = 0; i < strlen(optarg); i++)
857  optarg[i] = (char)tolower((unsigned char)optarg[i]);
858 
859  if (strcmp(optarg, "p2a") != 0 && strcmp(optarg, "p2h") != 0)
860  arg_error_exit("-, | --mode", optarg);
861 
862  // determine codec
863  sprintf(mode_description, "P2P + Encoding");
864 
865  if (optarg[2] == 'a')
866  {
867  *dst_codec_format = NI_CODEC_FORMAT_H264;
868  strcat(mode_description, " to AVC");
869  }
870 
871  if (optarg[2] == 'h')
872  {
873  *dst_codec_format = NI_CODEC_FORMAT_H265;
874  strcat(mode_description, " to HEVC");
875  }
876  printf("%s...\n", mode_description);
877 
878  break;
879  case 'o':
880  strcpy(output_filename, optarg);
881  break;
882  case 'r':
883  if (!(atoi(optarg) >= 1))
884  arg_error_exit("-r | --repeat", optarg);
885  g_repeat = atoi(optarg);
886  break;
887  default:
888  print_usage();
889  exit(1);
890  }
891  }
892 
893  // Check required args are present
894  if (!input_filename[0])
895  {
896  printf("Error: missing argument for -i | --input\n");
897  exit(-1);
898  }
899 
900  if (!output_filename[0])
901  {
902  printf("Error: missing argument for -o | --output\n");
903  exit(-1);
904  }
905 }
