128 void get_pcle_addr(char *device_name, char *pcle)
129 {
130  int i=0;
131  FILE *cmd_fp;
132  char *ptr = NULL;
133  char cmd[256] = {0};

135 
136  if(!device_name)
137  {

140  ptr = device_name + 5;
141  snprintf(cmd, sizeof(cmd) - 1, "udevadm info -q path -n %s | perl -nle'print $& "
142  "while m{(?<=/)[0-9a-f]{4}:[0-9a-f]{2}:[0-9a-f]{2}\\.[0-9a-f]}g' | tail -n 1",ptr);
143  cmd_fp = popen(cmd, "r");
144  if (!cmd_fp)
145  {
146  return ;
147  }

149  {
150  return ;
151  }
152  //copy cmd_ret to pcle and include 00.0
153  while(cmd_ret[i] && cmd_ret[i] != '\n')
