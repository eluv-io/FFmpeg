177 /*!*****************************************************************************
178  * \brief Whether SEI (HDR) should be sent together with this frame to encoder
179  *
180  * \param[in] p_enc_ctx encoder session context
181  * \param[in] pic_type frame type
182  * \param[in] p_param encoder parameters
183  *
184  * \return 1 if yes, 0 otherwise
185  ******************************************************************************/
186 int ni_should_send_sei_with_frame(ni_session_context_t *p_enc_ctx,
187  ni_pic_type_t pic_type,
188  ni_xcoder_params_t *p_param)
189 {
190  // repeatHeaders = 0. send on first frame only (IDR)
191  // repeatHeaders = 1. send on every I-frame including I-frames generated
192  // on the intraPeriod interval as well as I-frames that are forced.
193  if (0 == p_enc_ctx->frame_num || PIC_TYPE_IDR == pic_type ||
194  (p_param->cfg_enc_params.forced_header_enable &&
195  p_param->cfg_enc_params.intra_period &&
196  0 ==
197  ((p_enc_ctx->frame_num + p_enc_ctx->force_idr_intra_offset) %
198  p_param->cfg_enc_params.intra_period)))
199  {
200  if (PIC_TYPE_IDR == pic_type &&
201  p_param->cfg_enc_params.forced_header_enable &&
202  p_param->cfg_enc_params.intra_period &&
203  0 != (p_enc_ctx->frame_num % p_param->cfg_enc_params.intra_period))
204  {
205  p_enc_ctx->force_idr_intra_offset =
206  p_param->cfg_enc_params.intra_period -
207  (p_enc_ctx->frame_num % p_param->cfg_enc_params.intra_period);
208  }
209  ni_log(NI_LOG_TRACE, "should send sei? %lu %d %d %d %u\n",
210  p_enc_ctx->frame_num, pic_type,
211  p_param->cfg_enc_params.forced_header_enable,
212  p_param->cfg_enc_params.intra_period,
213  p_enc_ctx->force_idr_intra_offset);
214  return 1;
215  }
216  return 0;
217 }

226 /*!*****************************************************************************
227  * \brief Retrieve auxiliary data (close caption, various SEI) associated with
228  * this frame that is returned by decoder, convert them to appropriate
229  * format and save them in the frame's auxiliary data storage for
230  * future use by encoding. Usually they would be sent together with
231  * this frame to encoder at encoding.
232  *
233  * \param[in/out] frame that is returned by decoder
234  *
235  * \return NONE
236  ******************************************************************************/
237 void ni_dec_retrieve_aux_data(ni_frame_t *frame)
238 {
239  uint8_t *sei_buf = NULL;
240  ni_aux_data_t *aux_data = NULL;
241  int start_offset = 0;
242  if (frame->data_len[3] != 0)
243  {
244  //HW frame
245  start_offset = 3;
246  }
247  // User Data Unregistered SEI if available
248  if (frame->sei_user_data_unreg_len && frame->sei_user_data_unreg_offset)
249  {
250  sei_buf = (uint8_t *)frame->p_data[start_offset] +
251  frame->sei_user_data_unreg_offset;
252  if (!ni_frame_new_aux_data_from_raw_data(
253  frame, NI_FRAME_AUX_DATA_UDU_SEI, sei_buf,
254  (int)frame->sei_user_data_unreg_len))
255  {
256  ni_log(NI_LOG_ERROR, "ni_dec_retrieve_aux_data error retrieve User data "
257  "unregisted SEI!\n");
258  }
259  }
260 
261  // close caption data if available
262  if (frame->sei_cc_len && frame->sei_cc_offset)
263  {
264  sei_buf = (uint8_t *)frame->p_data[start_offset] + frame->sei_cc_offset;
265  if (!ni_frame_new_aux_data_from_raw_data(
266  frame, NI_FRAME_AUX_DATA_A53_CC, sei_buf,
267  (int)frame->sei_cc_len))
268  {
269  ni_log(NI_LOG_ERROR, "ni_dec_retrieve_aux_data error retrieve close "
270  "caption SEI!\n");
271  }
272  }
273 
274  // hdr10 sei data if available
275 
276  // mastering display metadata
277  if (frame->sei_hdr_mastering_display_color_vol_len &&
278  frame->sei_hdr_mastering_display_color_vol_offset)
279  {
280  aux_data = ni_frame_new_aux_data(
281  frame, NI_FRAME_AUX_DATA_MASTERING_DISPLAY_METADATA,
282  sizeof(ni_mastering_display_metadata_t));
283 
284  if (!aux_data)
285  {
286  ni_log(NI_LOG_ERROR, "ni_dec_retrieve_aux_data error retrieve HDR10 "
287  "mastering display color SEI!\n");
288  } else
289  {
290  ni_mastering_display_metadata_t *mdm =
291  (ni_mastering_display_metadata_t *)aux_data->data;
292  const int chroma_den = MASTERING_DISP_CHROMA_DEN;
293  const int luma_den = MASTERING_DISP_LUMA_DEN;
294  ni_dec_mastering_display_colour_volume_bytes_t *pColourVolume =
295  (ni_dec_mastering_display_colour_volume_bytes_t
296  *)((uint8_t *)frame->p_data[start_offset] +
297  frame->sei_hdr_mastering_display_color_vol_offset);
298 
299  // HEVC uses a g,b,r ordering, which we convert to a more natural r,
300  // g,b,this is so we are compatible with FFmpeg default soft decoder
301  mdm->display_primaries[0][0].num =
302  ntohs(pColourVolume->display_primaries[2][0]);
303  mdm->display_primaries[0][0].den = chroma_den;
304  mdm->display_primaries[0][1].num =
305  ntohs(pColourVolume->display_primaries[2][1]);
306  mdm->display_primaries[0][1].den = chroma_den;
307  mdm->display_primaries[1][0].num =
308  ntohs(pColourVolume->display_primaries[0][0]);
309  mdm->display_primaries[1][0].den = chroma_den;
310  mdm->display_primaries[1][1].num =
311  ntohs(pColourVolume->display_primaries[0][1]);
312  mdm->display_primaries[1][1].den = chroma_den;
313  mdm->display_primaries[2][0].num =
314  ntohs(pColourVolume->display_primaries[1][0]);
315  mdm->display_primaries[2][0].den = chroma_den;
316  mdm->display_primaries[2][1].num =
317  ntohs(pColourVolume->display_primaries[1][1]);
318  mdm->display_primaries[2][1].den = chroma_den;
319  mdm->white_point[0].num = ntohs(pColourVolume->white_point_x);
320  mdm->white_point[0].den = chroma_den;
321  mdm->white_point[1].num = ntohs(pColourVolume->white_point_y);
322  mdm->white_point[1].den = chroma_den;
323 
324  mdm->min_luminance.num =
325  ntohl(pColourVolume->min_display_mastering_luminance);
326  mdm->min_luminance.den = luma_den;
327  mdm->max_luminance.num =
328  ntohl(pColourVolume->max_display_mastering_luminance);
329  mdm->max_luminance.den = luma_den;
330 
331  mdm->has_luminance = mdm->has_primaries = 1;
332  }
333  }
334 
335  // hdr10 content light level
336  if (frame->sei_hdr_content_light_level_info_len &&
337  frame->sei_hdr_content_light_level_info_offset)
338  {
339  aux_data =
340  ni_frame_new_aux_data(frame, NI_FRAME_AUX_DATA_CONTENT_LIGHT_LEVEL,
341  sizeof(ni_content_light_level_t));
342 
343  if (!aux_data)
344  {
345  ni_log(NI_LOG_ERROR, "ni_dec_retrieve_aux_data error retrieve HDR10 "
346  "content light level SEI !\n");
347  } else
348  {
349  ni_content_light_level_t *clm =
350  (ni_content_light_level_t *)aux_data->data;
351  ni_content_light_level_info_bytes_t *pLightLevel =
352  (ni_content_light_level_info_bytes_t
353  *)((uint8_t *)frame->p_data[start_offset] +
354  frame->sei_hdr_content_light_level_info_offset);
355 
356  clm->max_cll = ntohs(pLightLevel->max_content_light_level);
357  clm->max_fall = ntohs(pLightLevel->max_pic_average_light_level);
358  }
359  }
360 
361  // hdr10+ sei data if available
362  if (frame->sei_hdr_plus_len && frame->sei_hdr_plus_offset)
363  {
364  aux_data = ni_frame_new_aux_data(frame, NI_FRAME_AUX_DATA_HDR_PLUS,
365  sizeof(ni_dynamic_hdr_plus_t));
366 
367  if (!aux_data)
368  {
369  ni_log(NI_LOG_ERROR, "ni_dec_retrieve_aux_data error retrieve HDR10+ SEI "
370  "!\n");
371  } else
372  {
373  int w, i, j, i_limit, j_limit;
374  ni_dynamic_hdr_plus_t *hdrp =
375  (ni_dynamic_hdr_plus_t *)aux_data->data;
376  ni_bitstream_reader_t br;
377 
378  sei_buf = (uint8_t *)frame->p_data[start_offset] +
379  frame->sei_hdr_plus_offset;
380  ni_bitstream_reader_init(&br, sei_buf,
381  8 * (int)frame->sei_hdr_plus_len);
382  hdrp->itu_t_t35_country_code = 0xB5;
383  hdrp->application_version = 0;
384  // first 7 bytes of t35 SEI data header already matched HDR10+, and:
385  ni_bs_reader_skip_bits(&br, 7 * 8);
386 
387  // Num_windows u(2)
388  hdrp->num_windows = ni_bs_reader_get_bits(&br, 2);
389  ni_log(NI_LOG_DEBUG, "hdr10+ num_windows %u\n", hdrp->num_windows);
390  if (!(1 == hdrp->num_windows || 2 == hdrp->num_windows ||
391  3 == hdrp->num_windows))
392  {
393  // wrong format and skip this HDR10+ SEI
394  } else
395  {
396  // the following block will be skipped for hdrp->num_windows ==
397  // 1
398  for (w = 1; w < hdrp->num_windows; w++)
399  {
400  hdrp->params[w - 1].window_upper_left_corner_x =
401  ni_make_q(ni_bs_reader_get_bits(&br, 16), 1);
402  hdrp->params[w - 1].window_upper_left_corner_y =
403  ni_make_q(ni_bs_reader_get_bits(&br, 16), 1);
404  hdrp->params[w - 1].window_lower_right_corner_x =
405  ni_make_q(ni_bs_reader_get_bits(&br, 16), 1);
406  hdrp->params[w - 1].window_lower_right_corner_y =
407  ni_make_q(ni_bs_reader_get_bits(&br, 16), 1);

409  ni_bs_reader_get_bits(&br, 16);

411  ni_bs_reader_get_bits(&br, 16);
412  hdrp->params[w - 1].rotation_angle =
413  ni_bs_reader_get_bits(&br, 8);

423  }
424 
425  // values are scaled down according to standard spec
426  hdrp->targeted_system_display_maximum_luminance.num =
427  ni_bs_reader_get_bits(&br, 27);
428  hdrp->targeted_system_display_maximum_luminance.den = 10000;
429 
430  hdrp->targeted_system_display_actual_peak_luminance_flag =
431  ni_bs_reader_get_bits(&br, 1);
432 
433  ni_log(NI_LOG_DEBUG,
434  "hdr10+ targeted_system_display_maximum_luminance "
435  "%d\n",
436  hdrp->targeted_system_display_maximum_luminance.num);
437  ni_log(NI_LOG_DEBUG,
438  "hdr10+ targeted_system_display_actual_peak_lumi"
439  "nance_flag %u\n",
440  hdrp->targeted_system_display_actual_peak_luminance_flag);
441 
442  if (hdrp->targeted_system_display_actual_peak_luminance_flag)
443  {
444  i_limit =
445  hdrp->num_rows_targeted_system_display_actual_peak_luminance =
446  ni_bs_reader_get_bits(&br, 5);
447 
448  j_limit =
449  hdrp->num_cols_targeted_system_display_actual_peak_luminance =
450  ni_bs_reader_get_bits(&br, 5);
451 
452  ni_log(NI_LOG_DEBUG,
453  "hdr10+ num_rows_targeted_system_display_actual"
454  "_peak_luminance x "
455  "num_cols_targeted_system_display_actual_"
456  "peak_luminance %d x %d\n",
457  i_limit, j_limit);
458 
459  i_limit = i_limit > 25 ? 25 : i_limit;
460  j_limit = j_limit > 25 ? 25 : j_limit;
461  for (i = 0; i < i_limit; i++)
462  for (j = 0; j < j_limit; j++)
463  {
464  hdrp->targeted_system_display_actual_peak_luminance
465  [i][j]
466  .num = ni_bs_reader_get_bits(&br, 4);
467  hdrp->targeted_system_display_actual_peak_luminance
468  [i][j]
469  .den = 15;
470  ni_log(NI_LOG_DEBUG,
471  "hdr10+ targeted_system_display_actual_peak"
472  "_luminance[%d][%d] %d\n",
473  i, j,
474  hdrp->targeted_system_display_actual_peak_luminance
475  [i][j]
476  .num);
477  }
478  }
479 
480  for (w = 0; w < hdrp->num_windows; w++)
481  {
482  for (i = 0; i < 3; i++)
483  {
484  hdrp->params[w].maxscl[i].num =
485  ni_bs_reader_get_bits(&br, 17);
486  hdrp->params[w].maxscl[i].den = 100000;
487  ni_log(NI_LOG_DEBUG, "hdr10+ maxscl[%d][%d] %d\n", w, i,
488  hdrp->params[w].maxscl[i].num);
489  }
490  hdrp->params[w].average_maxrgb.num =
491  ni_bs_reader_get_bits(&br, 17);

679 // internal function to convert struct of ROIs to NetInt ROI map and store
680 // them inside the encoder context passed in.
681 // return 0 if successful, -1 otherwise
682 static int set_roi_map(ni_session_context_t *p_enc_ctx,
683  ni_codec_format_t codec_format,
684  const ni_aux_data_t *aux_data, int nb_roi, int width,
685  int height, int intra_qp)
686 {
687  int r;
688  uint32_t i, j, k, m;
689  const ni_region_of_interest_t *roi =
690  (const ni_region_of_interest_t *)aux_data->data;
691  uint32_t self_size = roi->self_size;
692  int32_t set_qp = 0;
693  uint32_t sumQp = 0;
694 
695  uint32_t max_cu_size = (codec_format == NI_CODEC_FORMAT_H264) ? 16 : 64;
696 
697  // AV1 non-8x8-aligned resolution is implicitly cropped due to Quadra HW limitation
698  if (NI_CODEC_FORMAT_AV1 == codec_format)
699  {
700  width = (width / 8) * 8;
701  height = (height / 8) * 8;
702  }
703 
704  // for H.264, select ROI Map Block Unit Size: 16x16
705  // for H.265, select ROI Map Block Unit Size: 64x64
706  uint32_t roiMapBlockUnitSize =
707  (codec_format == NI_CODEC_FORMAT_H264) ? 16 : 64;
708  uint32_t mbWidth = ((width + max_cu_size - 1) & (~(max_cu_size - 1))) /
709  roiMapBlockUnitSize;
710  uint32_t mbHeight = ((height + max_cu_size - 1) & (~(max_cu_size - 1))) /
711  roiMapBlockUnitSize;
712  uint32_t numMbs = mbWidth * mbHeight;
713  uint32_t subMbWidth = roiMapBlockUnitSize / 8;
714  uint32_t subMbHeight = subMbWidth;
715  uint32_t subNumMbs = subMbWidth * subMbHeight;
716 
717  // (ROI map version >= 1) each QP info takes 8-bit, represent 8x8 pixel
718  // block
719  uint32_t block_size = ((width + max_cu_size - 1) & (~(max_cu_size - 1))) *
720  ((height + max_cu_size - 1) & (~(max_cu_size - 1))) / (8 * 8);
721 
722  // need to align to 64 bytes
723  uint32_t customMapSize = ((block_size + 63) & (~63));
724  if (!p_enc_ctx->roi_map)
725  {
726  p_enc_ctx->roi_map =
727  (ni_enc_quad_roi_custom_map *)calloc(1, customMapSize);
728  if (!p_enc_ctx->roi_map)
729  {
730  return -1;
731  }
732  }
733 
734  // init ipcm_flag to 0, roiAbsQp_falg to 0 (qp delta), and qp_info to 0
735  memset(p_enc_ctx->roi_map, 0, customMapSize);
736 
737  // iterate ROI list from the last as regions are defined in order of
738  // decreasing importance.
739  for (r = nb_roi - 1; r >= 0; r--)
740  {
741  roi = (const ni_region_of_interest_t *)(aux_data->data + self_size * r);
742  if (!roi->qoffset.den)
743  {
744  ni_log(NI_LOG_DEBUG, "ni_region_of_interest_t.qoffset.den must not be "
745  "zero.\n");
746  continue;
747  }
748 
749  set_qp = (int32_t)((float)roi->qoffset.num * 1.0f /
750  (float)roi->qoffset.den * NI_INTRA_QP_RANGE);
751  set_qp = clip3(NI_MIN_QP_DELTA, NI_MAX_QP_DELTA, set_qp);
752  // Adjust qp delta range (-25 to 25) to (0 to 63): 0 to 0, -1 to 1, -2 to
753  // 2 ... 1 to 63, 2 to 62 ...
754  // Theoretically the possible qp delta range is (-32 to 31)
755  set_qp = (NI_MAX_QP_INFO + 1 - set_qp) % (NI_MAX_QP_INFO + 1);
756 
757  ni_log(NI_LOG_DEBUG,
758  "set_roi_map: left %d right %d top %d bottom %d num %d den"
759  " %d set_qp %d\n",
760  roi->left, roi->right, roi->top, roi->bottom, roi->qoffset.num,
761  roi->qoffset.den, set_qp);
762 
763  // copy ROI MBs QPs into custom map
764  for (j = 0; j < mbHeight; j++)
765  {
766  for (i = 0; i < mbWidth; i++)
767  {
768  k = j * (int)mbWidth + i;
769 
770  for (m = 0; m < subNumMbs; m++)
771  {
772  if (((int)(i % mbWidth) >=
773  (int)((roi->left + roiMapBlockUnitSize - 1) /
774  roiMapBlockUnitSize) -
775  1) &&
776  ((int)(i % mbWidth) <=
777  (int)((roi->right + roiMapBlockUnitSize - 1) /
778  roiMapBlockUnitSize) -
779  1) &&
780  ((int)(j % mbHeight) >=
781  (int)((roi->top + roiMapBlockUnitSize - 1) /
782  roiMapBlockUnitSize) -
783  1) &&
784  ((int)(j % mbHeight) <=
785  (int)((roi->bottom + roiMapBlockUnitSize - 1) /
786  roiMapBlockUnitSize) -
787  1))
788  {
789  p_enc_ctx->roi_map[k * subNumMbs + m].field.ipcm_flag =
790  0; // don't force skip mode
791  p_enc_ctx->roi_map[k * subNumMbs + m]
792  .field.roiAbsQp_flag = 0; // delta QP
793  p_enc_ctx->roi_map[k * subNumMbs + m].field.qp_info =
794  set_qp;
795  // ni_log(NI_LOG_DEBUG, "## x %d y %d index %d\n", i,
796  // j, k*subNumMbs+m);
797  }
798  }
799  sumQp += p_enc_ctx->roi_map[k * subNumMbs].field.qp_info;
800  }
801  }
802  }
803 
804  p_enc_ctx->roi_len = customMapSize;
805  p_enc_ctx->roi_avg_qp =
806  (sumQp + (numMbs >> 1)) / numMbs + NI_DEFAULT_INTRA_QP; // round off
807 
808  return 0;
809 }
