297  * \param

299  * \return On success returns 0
300  * On failure returns <0
301  *******************************************************************************/
302 static inline int ni_pthread_mutex_init(ni_pthread_mutex_t *mutex)
303 {
304 #ifdef _WIN32
305  InitializeCriticalSection(mutex);
306  return 0;
307 #else
308  int rc;
309  ni_pthread_mutexattr_t attr;
310 
311  rc = pthread_mutexattr_init(&attr);
312  if (rc != 0)
313  {
314  return -1;
315  }
316 
317  /* For some cases to prevent the lock owner locking twice (i.e. internal
318  * API calls or if user application decides to lock the xcoder_mutex outside
319  * of API), The recursive mutex is a nice thing to solve the problem.
320  */
321  pthread_mutexattr_settype(&attr, PTHREAD_MUTEX_RECURSIVE);
322 
323  return pthread_mutex_init(mutex, &attr);
324 #endif
325 }

345 /*!******************************************************************************
346  * \brief thread mutex alloc&init
347  *
348  * \param
349  *
350  * \return On success returns true
351  * On failure returns false
352  *******************************************************************************/
353 static bool ni_pthread_mutex_alloc_and_init(ni_pthread_mutex_t **mutex)
354 {
355  int rc = 0;
356  *mutex = (ni_pthread_mutex_t *)calloc(1, sizeof(ni_pthread_mutex_t));
357  if (!(*mutex))
358  {
359  return false;
360  }
361 
362  rc = ni_pthread_mutex_init(*mutex);
363  if (rc != 0)
364  {
365  free(*mutex);
366  return false;
367  }
368 
369  return true;
370 }

372 /*!******************************************************************************
373  * \brief thread mutex free&destroy
374  *
375  * \param
376  *
377  * \return On success returns true
378  * On failure returns false
379  *******************************************************************************/
380 static bool ni_pthread_mutex_free_and_destroy(ni_pthread_mutex_t **mutex)
381 {
382  int rc = 0;
383  void *p;
384  static void *const tmp = NULL;
385 
386  // Avoid static analyzer inspection
387  memcpy(&p, mutex, sizeof(p));
388  if (p != NULL)
389  {
390  rc = ni_pthread_mutex_destroy((ni_pthread_mutex_t *)p);
391  memcpy(mutex, &tmp, sizeof(p));
392  free(p);
393  } else
394  {
395  rc = -1;
396  }
397 
398  return rc == 0;
399 }
