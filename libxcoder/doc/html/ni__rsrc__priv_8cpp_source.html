249 /*!*****************************************************************************
250  * \brief Check if a FW_rev retrieved from card is supported by libxcoder.
251  * Support is checked for API flavor, API version, and SW/FW release version.
252  *
253  * \param[in] fw_rev FW revision queried from card firmware
254  *
255  * \return If FW is fully compatible return 1
256  * If FW not compatible return 0
257  * If FW is partially compatible return 2
258  ******************************************************************************/
259 int ni_is_fw_compatible(uint8_t fw_rev[8])
260 {
261  char str_fw_ver[4], str_fw_API_fla[3], str_fw_API_ver[3];
262  int fw_API_ver;
263  char *strtok_token;
264  char str_fw_API_flav_support[] = NI_XCODER_FW_API_FLAVORS_SUPPORTED;
265  const char str_API_flavor_list_delim[2] = ",";
266  int api_flavor_matched = 0;
267 
268  // extract fw version
269  memcpy(str_fw_ver, fw_rev, NI_XCODER_VER_SZ);
270  str_fw_ver[NI_XCODER_VER_SZ] = '\0';
271 
272  // extract fw API flavor
273  memcpy(str_fw_API_fla, &(fw_rev[NI_XCODER_VER_SZ + 1]), NI_XCODER_API_FLAVOR_SZ);
274  str_fw_API_fla[NI_XCODER_API_FLAVOR_SZ] = '\0';
275 
276  // extract fw API version
277  memcpy(str_fw_API_ver, &(fw_rev[NI_XCODER_VER_SZ + 1 + NI_XCODER_API_FLAVOR_SZ]),
278  NI_XCODER_API_VER_SZ);
279  str_fw_API_ver[NI_XCODER_API_VER_SZ] = '\0';
280 
281  fw_API_ver = atoi(str_fw_API_ver);
282 
283  // check multiple API flavor support
284  strtok_token = strtok(str_fw_API_flav_support, str_API_flavor_list_delim);
285  while(strtok_token != NULL)
286  {
287  if (0 == strcmp(str_fw_API_fla, strtok_token))
288  {
289  api_flavor_matched = 1;
290  break;
291  }
292  strtok_token = strtok(NULL, str_API_flavor_list_delim);
293  }
294 
295  if (! (api_flavor_matched &&
296  fw_API_ver >= NI_XCODER_FW_API_VER_SUPPORTED_MIN))
297  {
298  /* ni_log(NI_LOG_INFO, "API-flavor: %s, API-version: %s, is not supported by "
299  "this libxcoder version: %s\n", str_fw_API_fla,
300  str_fw_API_ver, NI_XCODER_REVISION); */
301  return 0; // FWrev is not compatible
302  }
303  else if (strncmp(str_fw_ver, NI_XCODER_FW_VER_SUPPORTED_MIN, NI_XCODER_VER_SZ)
304  < 0)
305  {
306  /* ni_log(NI_LOG_INFO, "WARNING - FW version: %s is below the minimum support version "
307  "(%s) of this libxcoder: %s\n", str_fw_ver,
308  NI_XCODER_FW_VER_SUPPORTED_MIN, NI_XCODER_REVISION); */
309  return 2; // FWrev compatability supports core features but may not support some extra features
310  }
311  return 1; // FWrev is fully compatible
312 }
