177 void listModuleId(void)
178 {
179  /* read back the stored coders numbers */
180  uint32_t i;
181  ni_device_queue_t *ptr;
182  ni_device_pool_t *p_device_pool;
183  ni_device_info_t *p_device_info;
184 
185  p_device_pool = ni_rsrc_get_device_pool();
186  if (p_device_pool)
187  {
188 #ifdef _WIN32
189  if (WAIT_ABANDONED == WaitForSingleObject(p_device_pool->lock, INFINITE)) // no time-out interval)
190  {
191  fprintf(stderr, "ERROR: listModuleId() failed o obtain mutex: %p", p_device_pool->lock);
192  return;
193  }
194 #elif __linux__
195  if( flock(p_device_pool->lock, LOCK_EX) )
196  {
197  fprintf(stderr, "Error flock() failed\n");
198  }
199 #endif
200 
201  /* print out coders in their current order */
202  ptr = p_device_pool->p_device_queue;
203  printf("Num decoders: %d\n", ptr->decoders_cnt);
204  for (i = 0; i < ptr->decoders_cnt; i++) {
205  p_device_info = ni_rsrc_get_device_info(NI_DEVICE_TYPE_DECODER, ptr->decoders[i]);
206  printf("[%d]. %d (load: %d inst: %d %s.%d)\n", i, ptr->decoders[i],
207  p_device_info->load, p_device_info->active_num_inst, p_device_info->dev_name, p_device_info->hw_id);
208  free(p_device_info);
209  }
210  printf("\n\nNum encoders: %d\n", ptr->encoders_cnt);
211  for (i = 0; i < ptr->encoders_cnt; i++) {
212  p_device_info = ni_rsrc_get_device_info(NI_DEVICE_TYPE_ENCODER, ptr->encoders[i]);
213  printf("[%d]. %d (load: %d inst: %d %s.%d)\n", i, ptr->encoders[i],
214  p_device_info->load, p_device_info->active_num_inst, p_device_info->dev_name, p_device_info->hw_id);
215  free(p_device_info);
216  }
217  printf("\n\n");
218 
219 #ifdef _WIN32
220  ReleaseMutex(p_device_pool->lock);
221 #elif __linux__
222  if( flock(p_device_pool->lock, LOCK_UN) )
223  {
224  fprintf(stderr, "Error flock() failed\n");
225  }
226 #endif
227  ni_rsrc_free_device_pool(p_device_pool);
228  }
229 }
