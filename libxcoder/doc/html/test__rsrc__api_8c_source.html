201 
202  /* print out coders in their current order */
203  ptr = p_device_pool->p_device_queue;
204  for (k = 0; k < NI_DEVICE_TYPE_XCODER_MAX; k++)

207  for (i = 0; i < ptr->xcoder_cnt[k]; i++)

224  {
225  fprintf(stderr, "Error lockf() failed\n");
226  }
227 #endif
228  ni_rsrc_free_device_pool(p_device_pool);
229  }

406 static void displayRsrcMon(void)
407 {
408  ni_device_pool_t *p_device_pool = NULL;
409  ni_device_queue_t *coders = NULL;
410  ni_session_context_t xCtxt = {0};
411  int k;
412 
413  p_device_pool = ni_rsrc_get_device_pool();
414  if (!p_device_pool)
415  {
416  fprintf(stderr, "FATAL: cannot get devices info\n");
417  return;
418  }
419 
420  printf("**************************************************\n");
421  ni_device_session_context_init(&xCtxt);
422  coders = p_device_pool->p_device_queue;
423 
424  for (k = 0; k < NI_DEVICE_TYPE_XCODER_MAX; k++)
425  {
426  ni_device_type_t module_type = k;
427  ni_session_context_t *sessionCtxt = &xCtxt;
428  int i; // used in later FOR-loop when compiled without c99
429  int module_count = 0;
430  char module_name[8] = {'\0'};
431  int32_t *module_id_arr = NULL;
432  ni_device_context_t *p_device_context = NULL;
433 
434  if (!IS_XCODER_DEVICE_TYPE(module_type))
435  {
436  fprintf(stderr, "ERROR: unsupported module_type %d\n", module_type);
437  break;
438  }
439 
440  module_count = coders->xcoder_cnt[module_type];
441  strcpy(module_name, g_device_type_str[module_type]);
442 
443  module_id_arr = malloc(sizeof(*module_id_arr) * module_count);
444  if (!module_id_arr)
445  {
446  fprintf(stderr, "ERROR: malloc() failed for module_id_arr\n");
447  break;
448  }
449  memcpy(module_id_arr, coders->xcoders[module_type],
450  sizeof(int32_t) * module_count);
451 
452  printf("Num %ss: %d\n", g_device_type_str[module_type], module_count);
453 
454  // gotta test how long this sorting takes (SZ)
455  // sort module IDs used
456  qsort(module_id_arr, module_count, sizeof(int32_t), compareInt32_t);
457 
458  // Print performance info headings
459  if (IS_XCODER_DEVICE_TYPE(module_type))
460  {
461  printf("%-5s %-4s %-10s %-4s %-4s %-9s %-7s %-14s %-20s\n", "INDEX",
462  "LOAD", "MODEL_LOAD", "INST", "MEM", "SHARE_MEM", "P2P_MEM",
463  "DEVICE", "NAMESPACE");
464  }
465 
467  for (i = 0; i < module_count; i++)
468  {
469  p_device_context =
470  ni_rsrc_get_device_context(module_type, module_id_arr[i]);
471 
473  if (p_device_context)
474  {
475  sessionCtxt->device_handle =
476  ni_device_open(p_device_context->p_device_info->blk_name,
477  &sessionCtxt->max_nvme_io_size);
478  sessionCtxt->blk_io_handle = sessionCtxt->device_handle;
479 
480  // Check device can be opened
481  if (NI_INVALID_DEVICE_HANDLE == sessionCtxt->device_handle)
482  {
483  fprintf(stderr,
484  "ERROR: ni_device_open() failed for %s: %s\n",
485  p_device_context->p_device_info->blk_name,
486  strerror(NI_ERRNO));
487  ni_rsrc_free_device_context(p_device_context);
488  continue;
489  }
490 
491  sessionCtxt->hw_id = p_device_context->p_device_info->hw_id;
492  if (NI_RETCODE_SUCCESS !=
493  ni_device_session_query(sessionCtxt, module_type))
494  {
495  ni_device_close(sessionCtxt->device_handle);
496  fprintf(stderr, "Error query %s %s.%d\n", module_name,
497  p_device_context->p_device_info->dev_name,
498  p_device_context->p_device_info->hw_id);
499  ni_rsrc_free_device_context(p_device_context);
500  continue;
501  }
502  // printf("Done query %s %s.%d\n", module_name,
503  // p_device_context->p_device_info->dev_name,
504  // p_device_context->p_device_info->hw_id);
505  ni_device_close(sessionCtxt->device_handle);
506 
507  if (0 == sessionCtxt->load_query.total_contexts)
508  {
509  sessionCtxt->load_query.current_load = 0;
510  }
511 
512  // Print performance info row
513  if (IS_XCODER_DEVICE_TYPE(module_type))
514  {
515  printf("%-5d %-4u %-10u %-4u %-4u %-9u %-7u %-14s %-20s\n",
516  p_device_context->p_device_info->module_id,
517  sessionCtxt->load_query.current_load,
518  sessionCtxt->load_query.fw_model_load,
519  sessionCtxt->load_query.total_contexts,
520  sessionCtxt->load_query.fw_video_mem_usage,
521  sessionCtxt->load_query.fw_share_mem_usage,
522  sessionCtxt->load_query.fw_p2p_mem_usage,
523  p_device_context->p_device_info->dev_name,
524  p_device_context->p_device_info->blk_name);
525  }
526  ni_rsrc_free_device_context(p_device_context);
527  }
528  }
529  free(module_id_arr);
530  }
531 
532  ni_device_session_context_clear(&xCtxt);

535  printf("**************************************************\n");
