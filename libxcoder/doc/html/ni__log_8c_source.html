44 /*!*****************************************************************************
45  * \brief Get time for logs with microsecond timestamps
46  *
47  * \param[in/out] p_tp timeval struct
48  * \param[in] p_tzp timezone struct
49  *
50  * \return return 0 for success, -1 for error
51  ******************************************************************************/
52 int32_t ni_log_gettimeofday(struct timeval *p_tp, struct timezone *p_tzp)
53 {
54 #ifdef _WIN32
55  FILETIME file_time;
56  SYSTEMTIME system_time;
57  ULARGE_INTEGER ularge;
59  static const unsigned __int64 epoch =
60  ((unsigned __int64)116444736000000000ULL);
61 
62  // timezone information is stored outside the kernel so tzp isn't used
63  (void)p_tzp;
65  // Note: this function is not a precision timer. See elapsed_time().
66  GetSystemTime(&system_time);
67  SystemTimeToFileTime(&system_time, &file_time);
68  ularge.LowPart = file_time.dwLowDateTime;
69  ularge.HighPart = file_time.dwHighDateTime;
70  // Surpress cppcheck
71  (void)ularge.LowPart;
72  (void)ularge.HighPart;
73  p_tp->tv_sec = (long)((ularge.QuadPart - epoch) / 10000000L);
74  p_tp->tv_usec = (long)(system_time.wMilliseconds * 1000);
75 
76  return 0;
77 #else
78  return gettimeofday(p_tp, p_tzp);
79 #endif
80 }

82 /*!*****************************************************************************
83  * \brief Default ni_log() callback
84  *
85  * \param[in] level log level
86  * \param[in] fmt printf format specifier
87  * \param[in] vl variadric args list
88  *
89  * \return
90  ******************************************************************************/
91 void ni_log_default_callback(int level, const char* fmt, va_list vl)
92 {
93  if (level <= ni_log_level)
94  {
95 #ifndef _ANDROID
96 #ifdef NI_LOG_TRACE_TIMESTAMPS
97  if (level == NI_LOG_TRACE)
98  {
99  struct timeval tv;
100  ni_log_gettimeofday(&tv, NULL);
101 #ifdef _WIN32
102  fprintf(stderr, "[%lld] ", (long long) (tv.tv_sec * 1000000LL + tv.tv_usec));
103 #else
104  fprintf(stderr, "[%" PRIu64 "] ", (uint64_t) (tv.tv_sec * 1000000LL + tv.tv_usec));
105 #endif
106  }
107 #endif
108 #endif
109 
110 #ifdef _ANDROID
111  if (level == NI_LOG_INFO)
112  ALOGI(fmt, vl);
113  else if (level >= NI_LOG_ERROR)
114  ALOGE(fmt, vl);
115  else
116  ALOGV(fmt, vl);
117 #else
118  vfprintf(stderr, fmt, vl);
119 #endif
120  }
121 }

178 /*!*****************************************************************************
179  * \brief Convert ffmpeg log level integer to appropriate ni_log_level_t
180  *
181  * \param fflog_level integer representation of FFmpeg log level
182  *
183  * \return ni_log_level
184  ******************************************************************************/
185 ni_log_level_t ff_to_ni_log_level(int fflog_level)
186 {
187  ni_log_level_t converted_ni_log_level = NI_LOG_ERROR;
188  if (fflog_level <= -8)
189  {
190  converted_ni_log_level = NI_LOG_NONE;
191  }
192  else if (fflog_level <= 8)
193  {
194  converted_ni_log_level = NI_LOG_FATAL;
195  }
196  else if (fflog_level <= 16)
197  {
198  converted_ni_log_level = NI_LOG_ERROR;
199  }
200  else if (fflog_level <= 32)
201  {
202  converted_ni_log_level = NI_LOG_INFO;
203  }
204  else if (fflog_level <= 48)
205  {
206  converted_ni_log_level = NI_LOG_DEBUG;
207  }
208  else
209  {
210  converted_ni_log_level = NI_LOG_TRACE;
211  }
212  return converted_ni_log_level;
213 }

216 /*!*****************************************************************************

229 #endif
230 
231  if (!(*strcicmp)(arg_str, "none"))
232  {
233  return NI_LOG_NONE;
234  } else if (!(*strcicmp)(arg_str, "fatal"))
235  {
236  return NI_LOG_FATAL;
237  } else if (!(*strcicmp)(arg_str, "error"))
238  {
239  return NI_LOG_ERROR;
240  } else if (!(*strcicmp)(arg_str, "info"))
241  {
242  return NI_LOG_INFO;
243  } else if (!(*strcicmp)(arg_str, "debug"))
244  {
245  return NI_LOG_DEBUG;
246  } else if (!(*strcicmp)(arg_str, "trace"))
247  {
248  return NI_LOG_TRACE;
249  } else {
250  return NI_LOG_INVALID;
251  }
