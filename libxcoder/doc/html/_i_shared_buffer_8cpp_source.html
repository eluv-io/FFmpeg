51  /*!*****************************************************************************
52  * \brief Get file descripter by using android net.int.SharedBuffer service.
53  *
54  * \param[in] name The name of the share mem
55  *
56  * \return Get fd (> 0) , < 0 otherwise
57  *
58  ******************************************************************************/
59  int getFd(String8 &param)
60  {
61  Parcel data;
62  data.writeInterfaceToken(ISharedBuffer::getInterfaceDescriptor());
63 
64  Parcel reply;
65  data.writeString8(param);
66 
67  remote()->transact(GET_FD, data, &reply);
68 
69  if (reply.readInt32() == 1)
70  {
71  int fd = reply.readFileDescriptor();
72  int fd2 = dup(fd);
73  return fd2;
74  } else
75  {
76  return -1;
77  }
78  }

80  /*!*****************************************************************************

83  * \param[in] name The name of the share mem
84  *

86  *
87  ******************************************************************************/
88  int setFd(String8 &param, int32_t fd)
89  {
90  Parcel data;

116 status_t BnSharedBuffer::onTransact(uint32_t code, const Parcel &data,
117  Parcel *reply, uint32_t flags)
118 {
119  switch (code)
120  {
121  case GET_FD:
122  {
123  CHECK_INTERFACE(ISharedBuffer, data, reply);
124  String8 parma = data.readString8();
125  int fd = getFd(parma);
126  if (fd < 0)
127  {
128  reply->writeInt32(0);
129  return NO_ERROR;
130  }
131  reply->writeInt32(1);
132  reply->writeFileDescriptor(fd);
133  return NO_ERROR;
134  }
135  case SET_FD:
136  {
137  CHECK_INTERFACE(ISharedBuffer, data, reply);
138  String8 parma = data.readString8();
139  int32_t fd = data.readFileDescriptor();
140  int fd2 = dup(fd);
141  setFd(parma, fd2);
142  reply->writeInt32(0);
143  return NO_ERROR;
144  }
145  default:
146  {
147  return BBinder::onTransact(code, data, reply, flags);
148  }
149  }
150 }
